# final face-restoration pass in main_inference.py (pulls in PyTorch)
# NOTE: basicsr 1.4.2 imports torchvision.transforms.functional_tensor, removed in
# torchvision >=0.17. image_utils/face_restore.py installs a runtime shim for this,
# so no manual package patching is required.
#
# PyTorch is pinned to the CPU build. gfpgan/basicsr/facexlib otherwise pull the
# default GPU wheel from PyPI, whose bundled CUDA/NCCL conflicted with this machine
# (libtorch_cuda.so: undefined symbol ncclCommResume) and silently disabled GFPGAN.
# GFPGAN runs fine on CPU for single morph images, so we pin the +cpu wheels here.
--extra-index-url https://download.pytorch.org/whl/cpu
torch==2.12.1+cpu
torchvision==0.27.1+cpu
